  \resumeProject{Graph-Based Synthesis of Instrumental Variables for Confounded Treatment Effect Estimation}{Course Project}{\href{https://github.com/AmanPawar9/synthetic-instrumental-variables}{GitHub}}{Ongoing}
    \resumeItemListStart
        \item{Modeled the system as a graph of treatments, outcomes, and weak instruments, using a Graph Attention Network (GAT) to learn a synthetic instrumental variable embedding.}

    \resumeItemListEnd
  \resumeProject{Causal Discovery in Spatio-Temporal Systems via Self-Supervised Graph Neural Networks}{Course Project}{\href{https://github.com/AmanPawar9/causal-spatiotemporal-gnn}{GitHub}}{Ongoing}
    \resumeItemListStart
        \item{Represented complex spatio-temporal systems (e.g., climate patterns) as a dynamic graph, employing a Spatio-Temporal GNN (ST-GNN) to learn node representations that capture both local history and neighbor influence.}
        \item{Formulated a novel constraint-based causal discovery algorithm that operates in the learned ST-GNN embedding space, extending time-series methods to the more complex spatio-temporal domain.}
    \resumeItemListEnd
  \resumeProject{Uncovering Causal Gene Regulatory Networks with Biologically-Constrained Neural ODEs}{Course Project}{\href{https://github.com/AmanPawar9/causal-neural-ode-grn}{GitHub}}{Ongoing}
    \resumeItemListStart
      \item {Implemented a Neural Ordinary Differential Equation (Neural ODE) to model the continuous-time evolution of gene expression, enforcing biologically-plausible sparsity on the learned dynamics.}
      \item {Interpreted the learned sparse Jacobian of the system as a causal gene regulatory network, directly connecting the lab's work on ODEs with causal discovery in single-cell genomics.}
    \resumeItemListEnd
  \resumeProject{Graph-Informed Signature Kernels for Causal Discovery in Multivariate Clinical Time-Series}{Course Project}{\href{https://github.com/AmanPawar9/graph-informed-skci}{GitHub}}{Ongoing}
    \resumeItemListStart
      \item{Extended the state-of-the-art Signature Kernel Conditional Independence (SKCI) test by incorporating prior knowledge from biomedical knowledge graphs using GNN-based embeddings.}
      \item{Developed a novel, graph-informed signature kernel to achieve more accurate and data-efficient causal discovery from complex, multi-variate time-series data (e.g., MIMIC-III).}
    \resumeItemListEnd
